            function computeStats() {
                let correct = 0;
                let incorrect = 0;
                let answered = 0;
                for (let i = 0; i < totalQ; i++) {
                    if (revealed[i]) {
                        answered++;
                        if (userAnswers[i] === QUESTIONS[i].correct) correct++;

            function updateStatsUI() {

                if (t.includes('ransomware')) return 'Ransomware';
                if (t.includes('attack surface')) return 'Attack Surface';
                if (t.includes('gdpr') || t.includes('ccpa')) return 'Regulations';
                if (t.includes('trojan')) return 'Trojan Horse';
                if (t.includes('firewall') || t.includes('ips') || t.includes('ids')) return 'Network Security';
                if (t.includes('supply chain')) return 'Supply Chain';
                if (t.includes('hacktivist')) return 'Hacktivism';
                if (t.includes('encryption')) return 'Encryption';
                return 'General';
            }

            // ── Render question ──
            function renderQuestion(index) {
                const q = QUESTIONS[index];
                if (!q) return;
                qNumber.textContent = 'Q' + (index + 1);
                qText.textContent = q.q;
                qTopic.textContent = '· ' + getTopic(q);
                const selectedIdx = userAnswers[index];
                const isRevealed = revealed[index];
                const letters = ['A', 'B', 'C', 'D'];
                const resultIcons = ['', ''];

                let html = '';
                q.opts.forEach((optText, i) => {
                    let classes = 'opt';
                    if (selectedIdx === i) classes += ' selected';
                    if (isRevealed) {
                        if (i === q.correct) classes += ' correct-reveal';
                        else if (selectedIdx === i && i !== q.correct) classes += ' wrong-reveal';
                        classes += ' disabled';
                    }
                    const checkedAttr = (selectedIdx === i) ? 'checked' : '';
                    let iconHtml = '';
                    if (isRevealed && i === q.correct) iconHtml =
                        '<span class="result-icon">✅</span>';
                    if (isRevealed && selectedIdx === i && i !== q.correct) iconHtml =
                        '<span class="result-icon">❌</span>';
                    html += `
                <div class="${classes}" data-optidx="${i}">
                  <label>
                    <input type="radio" name="question" value="${i}" ${checkedAttr} ${isRevealed ? 'disabled' : ''}>
                    <span class="letter-circle">${letters[i]}</span>
                    <span style="flex:1;">${optText}</span>
                    ${iconHtml}
                  </label>
                </div>
              `;
                });
                optionsContainer.innerHTML = html;

                // Feedback area
                if (isRevealed && selectedIdx !== null) {
                    const explanation = generateExplanation(q, selectedIdx);
                    feedbackArea.className = 'feedback-area visible ' + (explanation.isCorrect ? 'correct-fb' :
                        'incorrect-fb');
                    feedbackIcon.textContent = explanation.isCorrect ? '✅' : '❌';
                    feedbackTitle.textContent = explanation.title;
                    feedbackDetail.textContent = explanation.detail;
                    feedbackCorrectAnswer.style.display = 'none';
                    if (!explanation.isCorrect) {
                        feedbackCorrectAnswer.style.display = 'inline-block';
                        feedbackCorrectAnswer.textContent = 'Correct: ' + letters[q.correct] + '. ' + q.opts[q.correct];
                    }
                } else {
                    feedbackArea.className = 'feedback-area';
                    feedbackArea.style.display = 'none';
                }

                // Attach click listeners
                if (!isRevealed) {
                    document.querySelectorAll('.opt').forEach(optEl => {
                        optEl.addEventListener('click', function(e) {
                            const radio = this.querySelector('input[type="radio"]');
                            if (radio && !radio.disabled) {
                                radio.checked = true;
                                const val = parseInt(radio.value);
                                if (!isNaN(val)) {
                                    handleAnswerSelection(currentIndex, val);
                                }
                            }
                        });
                    });
                }

                // Update buttons
                prevBtn.disabled = (index === 0);
                if (index === totalQ - 1) {
                    nextBtn.textContent = 'Finish ▶';
                    nextBtn.className = 'primary-btn';
                } else {
                    nextBtn.textContent = 'Next ▶';
                    nextBtn.className = 'primary-btn';
                }
                const sel = userAnswers[index];
                infoMsg.textContent = isRevealed ? (sel === q.correct ? '✅ Correct answer' : '❌ Incorrect answer') :
                    (sel !== null ? '✓ Option selected (click to confirm)' : 'Select an option');
                updateStatsUI();
            }

            // ── Handle answer selection with immediate feedback ──
            function handleAnswerSelection(index, selectedValue) {
                if (revealed[index]) return; // Already revealed
                userAnswers[index] = selectedValue;
                revealed[index] = true; // Immediately reveal
                const q = QUESTIONS[index];
                const isCorrect = selectedValue === q.correct;

                // Flash animation
                qContainer.classList.remove('correct-flash', 'incorrect-flash');
                void qContainer.offsetWidth; // Trigger reflow
                if (isCorrect) {
                    qContainer.classList.add('correct-flash');
                } else {
                    qContainer.classList.add('incorrect-flash');
                }

                // Re-render to show feedback
                renderQuestion(index);
                updateStatsUI();

                // Scroll to feedback
                feedbackArea.scrollIntoView({ behavior: 'smooth', block: 'nearest' });
            }

            // ── Navigation ──
            function goTo(index) {
                if (index < 0) index = 0;
                if (index >= totalQ) index = totalQ - 1;
                currentIndex = index;
                renderQuestion(currentIndex);
                qContainer.scrollIntoView({ behavior: 'smooth', block: 'start' });
                feedbackArea.style.display = 'none';
                feedbackArea.className = 'feedback-area';
            }

            function finishQuiz() {
                const stats = computeStats();
                const unanswered = stats.remaining;
                if (unanswered > 0) {
                    if (!confirm(
                            'You have ' + unanswered +
                            ' unanswered question(s). Unanswered questions count as incorrect. Finish anyway?')) {
                        return;
                    }
                }
                // Reveal any remaining unanswered
                for (let i = 0; i < totalQ; i++) {
                    if (!revealed[i]) revealed[i] = true;
                }
                const finalStats = computeStats();
                const pct = Math.round((finalStats.correct / totalQ) * 100);
                quizArea.classList.add('hidden');
                resultsArea.classList.remove('hidden');
                document.getElementById('finalScoreNum').textContent = finalStats.correct;
                document.getElementById('finalPctLabel').textContent = pct + '%';
                document.getElementById('finalCorrect').textContent = finalStats.correct;
                document.getElementById('finalIncorrect').textContent = finalStats.incorrect;
                document.getElementById('finalUnanswered').textContent = finalStats.remaining;
                const scoreRing = document.getElementById('scoreRing');
                scoreRing.style.setProperty('--score-pct', pct);
                resultsArea.scrollIntoView({ behavior: 'smooth' });
            }

            function restartQuiz() {
                for (let i = 0; i < totalQ; i++) {
                    userAnswers[i] = null;
                    revealed[i] = false;
                }
                currentIndex = 0;
                resultsArea.classList.add('hidden');
                quizArea.classList.remove('hidden');
                feedbackArea.className = 'feedback-area';
                feedbackArea.style.display = 'none';
                qContainer.classList.remove('correct-flash', 'incorrect-flash');
                renderQuestion(0);
                updateStatsUI();
                window.scrollTo({ top: 0, behavior: 'smooth' });
            }

            // ── Event listeners ──
            prevBtn.addEventListener('click', () => goTo(currentIndex - 1));
            nextBtn.addEventListener('click', () => {
                if (currentIndex === totalQ - 1) {
                    finishQuiz();
                } else {
                    goTo(currentIndex + 1);
                }
            });
            finishBtn.addEventListener('click', () => finishQuiz());
            restartBtn.addEventListener('click', restartQuiz);

            // Keyboard navigation
            document.addEventListener('keydown', (e) => {
                if (resultsArea.classList.contains('hidden')) {
                    if (e.key === 'ArrowRight' || e.key === 'ArrowDown') {
                        e.preventDefault();
                        if (currentIndex < totalQ - 1) goTo(currentIndex + 1);
                        else if (currentIndex === totalQ - 1) finishQuiz();
                    } else if (e.key === 'ArrowLeft' || e.key === 'ArrowUp') {
                        e.preventDefault();
                        if (currentIndex > 0) goTo(currentIndex - 1);
                    } else if (e.key === 'Enter' && !revealed[currentIndex] && userAnswers[currentIndex] !== null) {
                        e.preventDefault();
                        handleAnswerSelection(currentIndex, userAnswers[currentIndex]);
                    }
                }
            });

            // Touch swipe support
            let touchStartX = 0;
            let touchStartY = 0;
            qContainer.addEventListener('touchstart', function(e) {
                touchStartX = e.touches[0].clientX;
                touchStartY = e.touches[0].clientY;
            }, { passive: true });
            qContainer.addEventListener('touchend', function(e) {
                const dx = e.changedTouches[0].clientX - touchStartX;
                const dy = e.changedTouches[0].clientY - touchStartY;
                if (Math.abs(dx) > Math.abs(dy) && Math.abs(dx) > 50) {
                    if (dx < -40 && currentIndex < totalQ - 1) {
                        goTo(currentIndex + 1);
                    } else if (dx > 40 && currentIndex > 0) {
                        goTo(currentIndex - 1);
                    }
                }
            });

            // ── Init ──
            renderQuestion(0);
            updateStatsUI();
        })();
    </script>
</body>
